\begin{frame}{项目概览}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.53\linewidth}
            \begin{itemize}
                \item \textbf{选题：}影响力最大化。
                \item \textbf{问题：}在网络中选择 $k$ 个种子节点，使期望扩散范围最大。
                \item \textbf{算法视角：}单调子模最大化、贪心近似、CELF 懒惰求值。
                \item \textbf{拓展：}GreeDI 风格的局部候选共享，并显式考虑通信预算。
            \end{itemize}
        \end{column}
        \begin{column}{0.43\linewidth}
            \metricbox{核心实验信息}{质量、运行时间和通信量之间存在真实权衡。}
            \metricbox{最清楚的算法收益}{CELF 在 large 设置下减少约 74\% 的 spread evaluations。}
        \end{column}
    \end{columns}

\begin{frame}{问题形式化}
    \[
        \max_{S\subseteq V,\ |S|\leq k} \sigma(S)
    \]
    \begin{itemize}
        \item $G=(V,E,p)$ 是带边激活概率 $p_{uv}$ 的有向图。
        \item $\sigma(S)$ 表示从种子集合 $S$ 出发后的期望激活节点数。
        \item 独立级联模型：
        \begin{itemize}
            \item 新激活节点对每个未激活出邻居只有一次激活机会；
            \item 每次激活尝试以概率 $p_{uv}$ 独立成功。
        \end{itemize}
        \item 实验采用 weighted-cascade 规则：$p_{uv}=1/\mathrm{indegree}(v)$。
    \end{itemize}
\end{frame}

\begin{frame}{为什么可以用贪心}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.52\linewidth}
            \begin{itemize}
                \item IC 模型下的期望影响范围是\textbf{单调}的：增加种子不会降低扩散范围。
                \item 它也是\textbf{子模}的：种子集合越大，新节点带来的边际收益越小。
            \end{itemize}
            \[
                \sigma(A\cup\{v\})-\sigma(A)
                \geq
                \sigma(B\cup\{v\})-\sigma(B)
            \]
            \[
                A\subseteq B,\quad v\notin B
            \]
        \end{column}
        \begin{column}{0.42\linewidth}
            \metricbox{近似保证}{精确目标下，greedy 有 $(1-1/e)$ 近似保证。}
            \metricbox{实际加速}{CELF 将旧边际收益作为当前上界来减少重算。}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{实现的方法}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.31\linewidth}
            \begin{block}{基线方法}
                \begin{itemize}
                    \item 随机种子
                    \item 加权出度
                    \item PageRank
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.31\linewidth}
            \begin{block}{中心化方法}
                \begin{itemize}
                    \item Monte Carlo greedy
                    \item CELF 懒惰贪心
                    \item 对最终种子集合公平再评估
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.31\linewidth}
            \begin{block}{分布式方法}
                \begin{itemize}
                    \item $m\in\{2,4,8\}$ 个分区
                    \item 本地预算 $q\in\{k,2k,5k\}$
                    \item 协调器在合并候选池上运行 greedy
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{结果解释与局限}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\linewidth}
            \begin{block}{已经完成的部分}
                \begin{itemize}
                    \item 复现了子模 greedy 的核心机制。
                    \item CELF 展示了清楚的实际加速效果。
                    \item 分布式候选共享在很多设置下接近 greedy 质量。
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\linewidth}
            \begin{block}{需要诚实说明的部分}
                \begin{itemize}
                    \item 课程规模模拟次数下，Monte Carlo 选择存在噪声。
                    \item 结构性基线在公平再评估后可能超过 greedy。
                    \item 当前分布式运行是顺序模拟；真正加速需要并行 worker。
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    \vspace{0.4em}
    \small 后续可做：更多随机种子、置信区间、更大真实网络、确定性划分，以及 RR-set 最大覆盖方法。
\end{frame}

\begin{frame}{结论}
    \begin{itemize}
        \item 影响力最大化很好地展示了经典算法如何服务现代网络数据问题。
        \item IC 模型下的影响范围是单调子模函数，因此 greedy 近似提供了理论主线。
        \item CELF 说明子模性不只是证明工具，也能转化为实际加速。
        \item GreeDI-style 候选共享把问题推进到通信受限的分布式优化场景。
        \item 关键经验不是某个方法永远最好，而是图结构、估计精度和通信预算共同决定效果。
    \end{itemize}
\end{frame}
