\begin{frame}{Problem Formulation}
    \[
        \max_{S\subseteq V,\ |S|\leq k} \sigma(S)
    \]
    \begin{itemize}
        \item $G=(V,E,p)$ is a directed graph with edge activation probability $p_{uv}$.
        \item $\sigma(S)$ is the expected number of active nodes after diffusion.
        \item Independent Cascade model:
        \begin{itemize}
            \item a newly active node gets one chance to activate each inactive out-neighbor;
            \item each activation attempt succeeds independently with probability $p_{uv}$.
        \end{itemize}
        \item Experiments use the weighted-cascade rule $p_{uv}=1/\mathrm{indegree}(v)$.
    \end{itemize}
\end{frame}

\begin{frame}{Why Greedy Applies}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.52\linewidth}
            \begin{itemize}
                \item IC expected spread is \textbf{monotone}: adding seeds cannot reduce spread.
                \item It is also \textbf{submodular}: marginal gain decreases as the seed set grows.
            \end{itemize}
            \[
                \sigma(A\cup\{v\})-\sigma(A)
                \geq
                \sigma(B\cup\{v\})-\sigma(B)
            \]
            \[
                A\subseteq B,\quad v\notin B
            \]
        \end{column}
        \begin{column}{0.42\linewidth}
            \metricbox{Approximation}{Greedy gives $(1-1/e)$ for the exact objective.}
            \metricbox{Acceleration}{CELF treats old marginal gains as upper bounds.}
        \end{column}
    \end{columns}
\end{frame}

    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.31\linewidth}
            \begin{block}{Baselines}
                \begin{itemize}
                    \item Random seeds
                    \item Weighted out-degree
                    \item PageRank
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.31\linewidth}
            \begin{block}{Centralized}
                \begin{itemize}
                    \item Monte Carlo greedy
                    \item CELF lazy greedy
                    \item Fair re-evaluation of final seed sets
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.31\linewidth}
            \begin{block}{Distributed}
                \begin{itemize}
                    \item $m\in\{2,4,8\}$ partitions
                    \item local budget $q\in\{k,2k,5k\}$
                    \item coordinator greedy on merged candidates
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Interpretation and Limitations}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\linewidth}
            \begin{block}{What worked}
                \begin{itemize}
                    \item Reproduced the submodular greedy mechanism.
                    \item CELF demonstrates a clear practical speedup.
                    \item Distributed candidate sharing stays close to greedy quality in many cases.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\linewidth}
            \begin{block}{What to state honestly}
                \begin{itemize}
                    \item Monte Carlo selection is noisy at course-scale simulation counts.
                    \item Structural baselines can beat greedy after fair re-evaluation.
                    \item Distributed runtime here is sequential; real speedup needs parallel workers.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    \vspace{0.4em}
    \small Next steps: more random seeds, confidence intervals, larger real networks, deterministic partitions, and RR-set based maximum coverage.
\end{frame}

\begin{frame}{Conclusion}
    \begin{itemize}
        \item Influence maximization is a clean example of classical algorithms serving a modern network-data problem.
        \item IC spread is monotone submodular, so greedy approximation gives the central theoretical anchor.
        \item CELF shows how submodularity becomes practical acceleration.
        \item GreeDI-style candidate sharing exposes the cost of limited communication.
        \item The key lesson is not that one method always wins: graph structure, estimation accuracy, and communication budget all matter.
    \end{itemize}
\end{frame}
